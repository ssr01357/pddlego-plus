  (:types object receptacle microwaveReceptacle sinkbasinReceptacle fridgeReceptacle sharpObject)
  (:predicates
    (at ?r - receptacle)
    (visited ?r - receptacle)
    (opened ?r - receptacle)

    (heated ?o - object)
    (switched-on ?o - object)
    (cleaned ?o - object)
    (cooled ?o - object)
    (sliced ?o - object)
  )

  (:action GotoLocation
    :parameters (?from - receptacle ?to - receptacle)
    :precondition (and (at ?from) (not (visited ?to)))
    :effect (and (not (at ?from)) (at ?to) (visited ?to))
  )

    :precondition (and (at ?r) (not (opened ?r)))

    :effect (not (opened ?r))
  )

  (:types 
    object
    receptacle - object
    microwaveReceptacle sinkbasinReceptacle fridgeReceptacle - receptacle
    sharpObject - object
  )
  (:predicates
    (at ?r - receptacle)
    (visited ?r - receptacle)
    (opened ?r - receptacle)
    (in ?o - object ?r - receptacle)
    (holding ?o - object)
    (heated ?o - object)
    (switched-on ?o - object)
    (cleaned ?o - object)
    (cooled ?o - object)
  )

  (:action GotoLocation
    :parameters (?from - receptacle ?to - receptacle)
    :precondition (and (at ?from) (not (visited ?to)))
    :effect (and (not (at ?from)) (at ?to) (visited ?to))
  )

  (:action OpenObject
    :parameters (?r - receptacle)
    :precondition (and (at ?r) (not (opened ?r)))
    :effect (opened ?r)
  )

  (:action CloseObject
    :parameters (?r - receptacle)
    :precondition (opened ?r)
    :effect (not (opened ?r))
  )

  (:action PickupObject
    :parameters (?o - object ?r - receptacle)
    :precondition (at ?r)
    :effect (and (holding ?o) (not (in ?o ?r)))
  )

  (:action PutObject
    :parameters (?o - object ?r - receptacle)
    :precondition (and (holding ?o) (at ?r))
    :effect (and (in ?o ?r) (not (holding ?o)))
  )

  (:action useObject
    :parameters (?o - object)
    :precondition (holding ?o)
    :effect (switched-on ?o)
  )

  (:action HeatObject
    :parameters (?o - object ?r - microwaveReceptacle)
    :precondition (and (holding ?o) (at ?r))
    :effect (heated ?o)
  )

  (:action CleanObject
    :parameters (?o - object ?r - sinkbasinReceptacle)
    :precondition (and (holding ?o) (at ?r))
    :effect (cleaned ?o)
  )

  (:action CoolObject
    :parameters (?o - object ?r - fridgeReceptacle)

  )
)
