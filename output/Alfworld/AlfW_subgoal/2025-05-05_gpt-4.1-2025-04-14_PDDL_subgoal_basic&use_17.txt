 (define (domain explore-room)
  (:requirements :strips :typing :negative-preconditions)
  (:types receptacle object)

  (:predicates
    (at ?r - receptacle)
    (visited ?r - receptacle)
    (open ?r - receptacle)
    (in ?o - object ?r - receptacle)
    (on ?o - object ?r - receptacle)
    (holding ?o - object)
    (used ?o - object)
  )

  (:action GotoLocation
    :parameters (?from - receptacle ?to - receptacle)
    :precondition (at ?from)
    :effect (and (at ?to) (visited ?to) (not (at ?from)))
  )

  (:action OpenObject
    :parameters (?r - receptacle)
    :precondition (and (at ?r) (not (open ?r)))
    :effect (open ?r)
  )

  (:action CloseObject
    :parameters (?r - receptacle)
    :precondition (and (at ?r) (open ?r))
    :effect (not (open ?r))
  )

  (:action PickupObject
    :parameters (?o - object ?r - receptacle)
    :precondition (and (at ?r) (or (in ?o ?r) (on ?o ?r)))
    :effect (and (holding ?o) (not (in ?o ?r)) (not (on ?o ?r)))
  )
