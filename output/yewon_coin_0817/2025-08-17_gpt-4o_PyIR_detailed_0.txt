class ExplorationProblem(ExplorationDomain):
    def __init__(self):
        super().__init__()
        self.locations = ExplorationDomain.Location.create_objs(["kitchen", "unknown_south", "unknown_west"])
        self.doors = ExplorationDomain.Door.create_objs(["patio_door", "plain_door"])

    @init
    def init(self):
        at = [self.at_location(self.locations["kitchen"])]
        closed_doors = [~self.door_open(door) for door in self.doors.values()]
        return at + closed_doors

    @goal
    def goal(self):
        return [self.at_location(self.locations["unknown_south"]),
                self.at_location(self.locations["unknown_west"])]
