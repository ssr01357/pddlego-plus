  (:requirements :typing :strips)

    (at ?l - location)

    (opened ?r - receptacle)
    (closed ?r - receptacle)
    (in ?o - object ?r - receptacle)
    (picked ?o - object)
  )

  (:action GotoLocation
    :parameters (?from - location ?to - location)
    :precondition (and (at ?from) (not (visited ?to)))
    :effect (and (not (at ?from)) (at ?to) (visited ?to))
  )

  (:action OpenObject
    :parameters (?r - receptacle)
