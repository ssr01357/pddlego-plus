class ExploreEnv(Domain):
    Location = create_type("Location")
    Door = create_type("Door")
    Direction = create_type("Direction")

    @predicate(Location)
    def visited(self, loc):
        """Whether a location has been visited"""

    @predicate(Location)
    def agent_at(self, loc):
        """Agent is at the given location"""

    @predicate(Door, Location, Location, Direction)
    def connects(self, door, loc1, loc2, direction):
        """Door connects loc1 to loc2 in given direction"""

    @predicate(Door)
    def is_open(self, door):
        """Door is open"""

    @predicate(Door)
    def is_closed(self, door):
        """Door is closed"""

    @predicate(Door, Location, Direction)
    def door_at(self, door, loc, direction):
        """Door is at the specified direction of the location"""

    @action(Door)
    def open_door(self, door):
        precond = [self.is_closed(door)]
        effect = [~self.is_closed(door), self.is_open(door)]
        return precond, effect

    @action(Door)
    def close_door(self, door):

        effect = [~self.is_open(door), self.is_closed(door)]

class ExploreEnv(Domain):
    Location = create_type("Location")
    Door = create_type("Door")
    Direction = create_type("Direction")

    @predicate(Location)
    def visited(self, loc):
        """Whether a location has been visited"""

    @predicate(Location)
    def agent_at(self, loc):
        """Agent is at the given location"""

    @predicate(Door, Location, Location, Direction)
    def connects(self, door, loc1, loc2, direction):
        """Door connects loc1 to loc2 in given direction"""

    @predicate(Door)
    def is_open(self, door):
        """Door is open"""

    @predicate(Door)
    def is_closed(self, door):
        """Door is closed"""

    @predicate(Door, Location, Direction)
    def door_at(self, door, loc, direction):
        """Door is at the specified direction of the location"""

    @action(Door)
    def open_door(self, door):
        precond = [self.is_closed(door)]
        effect = [~self.is_closed(door), self.is_open(door)]
        return precond, effect

    @action(Door)
    def close_door(self, door):
        precond = [self.is_open(door)]
        effect = [~self.is_open(door), self.is_closed(door)]
        return precond, effect
